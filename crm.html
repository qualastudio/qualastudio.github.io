DROP ALL EXISTING TABLES drop table if exists activities cascade; drop table if exists appointments cascade; drop table if exists deals cascade; drop table if exists invoices cascade; drop table if exists files cascade; drop table if exists clinical_records cascade; drop table if exists clients cascade; drop table if exists users cascade; drop table if exists businesses cascade; -- 2. BUSINESSES create table businesses ( id uuid default gen_random_uuid() primary key, name text not null, email text, phone text, website text, address text, logo_url text, -- Invoice settings invoice_prefix text default 'INV', invoice_next_number integer default 1, payment_terms text default 'Due on receipt', invoice_notes text default 'Thank you for your business!', -- Payment links stripe_link text, paypal_link text, mercadopago_link text, zelle_info text, venmo_info text, cashapp_info text, -- Preferences language text default 'en', theme text default 'light', currency text default 'USD', date_format text default 'MM/DD/YYYY', -- Module settings module_clinical boolean default false, created_at timestamp default now() ); -- 3. USERS (team members) create table users ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, auth_user_id uuid, full_name text not null, email text not null, role text default 'staff', -- admin, staff, readonly avatar_color text default '#3B82F6', is_active boolean default true, created_at timestamp default now() ); -- 4. CLIENTS create table clients ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, full_name text not null, company text, email text, phone text, whatsapp text, industry text, status text default 'active', total_value numeric default 0, source text, tags text[], address text, birthday date, notes text, is_favorite boolean default false, created_at timestamp default now() ); -- 5. INVOICES create table invoices ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete set null, invoice_number text, description text, items jsonb default '[]', subtotal numeric default 0, tax numeric default 0, amount numeric not null, currency text default 'USD', status text default 'pending', due_date date, paid_at timestamp, payment_method text, notes text, sent_at timestamp, created_at timestamp default now() ); -- 6. DEALS (pipeline) create table deals ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete set null, name text not null, value numeric default 0, currency text default 'USD', stage text default 'lead', probability integer default 20, expected_close date, notes text, created_at timestamp default now() ); -- 7. APPOINTMENTS create table appointments ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete set null, title text not null, description text, date date not null, time text, duration integer default 60, status text default 'scheduled', reminder_sent boolean default false, created_at timestamp default now() ); -- 8. FILES create table files ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete cascade, name text not null, url text not null, file_type text, size integer, uploaded_by text, created_at timestamp default now() ); -- 9. CLINICAL RECORDS (Cenepla / Medical module) create table clinical_records ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete cascade, date date not null, doctor text, type text default 'consultation', diagnosis text, treatment text, prescription text, notes text, study_results jsonb default '[]', follow_up_date date, created_at timestamp default now() ); -- 10. ACTIVITIES create table activities ( id uuid default gen_random_uuid() primary key, business_id uuid references businesses(id) on delete cascade, client_id uuid references clients(id) on delete set null, user_id uuid, type text, description text, metadata jsonb, created_at timestamp default now() ); -- 11.
